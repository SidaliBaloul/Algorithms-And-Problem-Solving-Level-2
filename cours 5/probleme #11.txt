}

int ReverseNumber(int num)
{
    int remainder = 0;
    int num2 = 0;
    while (num > 0)
    {
        remainder = num % 10;
        num /= 10;
        num2 = num2 * 10 + remainder;

    }
    return num2;
}
bool PrintReversed(int num1)
{
    return ReverseNumber(num1) == num1;
}
