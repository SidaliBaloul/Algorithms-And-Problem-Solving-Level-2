{
    int num;
    do
    {
        cout << msg << endl;

}

int ReverseNumber(int num1, short num2)
{
    int remainder = 0;
    int degits = 0;
    while (num1 > 0)
    {
        remainder = num1 % 10;
        num1 /= 10;

        if (remainder == num2)
            degits++;

        
    }
    return degits;
}
